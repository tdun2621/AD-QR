function Generate-UniquePassword($usedPasswords, $currentAdjectiveBank, $currentNounBank, $currentVerbBank) {
    $random = New-Object System.Random
    do {
        $verb = $currentVerbBank[$random.Next(0, $currentVerbBank.Count)]
        $adjective = $currentAdjectiveBank[$random.Next(0, $currentAdjectiveBank.Count)]
        $animal = $currentNounBank[$random.Next(0, $currentNounBank.Count)]
        $password = "$verb-$adjective-$animal"
    } while ($password.Length -lt 12)
    return $password
}

function ResizeTextBoxToContent {
    param(
        [System.Windows.Forms.TextBox]$TextBox,
        [int]$RightBoundaryX,
        [int]$MinimumWidth = 50,
        [int]$MaximumWidth = 300
    )
    if ([string]::IsNullOrEmpty($TextBox.Text)) {
        $TextBox.Width = $MinimumWidth
    } else {
        $measurementBounds = New-Object System.Drawing.Size($MaximumWidth, [int]::MaxValue)
        $textSize = [System.Windows.Forms.TextRenderer]::MeasureText($TextBox.Text, $TextBox.Font, $measurementBounds, [System.Windows.Forms.TextFormatFlags]::NoPadding)
        $requiredWidth = $textSize.Width + 15

        $availableWidth = $RightBoundaryX - $TextBox.Left
        
        $targetWidth = [System.Math]::Min($requiredWidth, $availableWidth)
        $targetWidth = [System.Math]::Min($targetWidth, $MaximumWidth)
        $TextBox.Width = [System.Math]::Max($targetWidth, $MinimumWidth)
    }
}

$buttonProcess.Add_Click({
    $dataGrid.Rows.Clear()
    $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - Starting AD query.`n")
    $buttonSave.Enabled = $false; SetButtonStateColors $buttonSave $false
    $buttonEnableReset.Enabled = $false; SetButtonStateColors $buttonEnableReset $false
    
    $lines = $textBoxUsers.Text -split '\r?\n' | Where-Object { -not [string]::IsNullOrWhiteSpace($_) }
    $totalLines = $lines.Count
    $progressBar.Maximum = $totalLines
    $progressBar.Value = 0
    $form.PerformLayout()

    foreach ($line in $lines) {
        $username = $line.Trim()
        if ([string]::IsNullOrWhiteSpace($username)) {
            $progressBar.Value++
            continue
        }
        
        try {
            $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - Querying user '$username'.`n")
            $user = Get-ADUser -Identity $username -Properties SamAccountName, LastLogonDate, Description, PhysicalDeliveryOfficeName, Enabled, mail, DistinguishedName -ErrorAction Stop
            
            $lastLogonDateProperty = $user.LastLogonDate
            $status = "Never"
            $lastLogonTimestamp = $null

            if ($lastLogonDateProperty) {
                if ($lastLogonDateProperty -is [datetime]) {
                    $lastLogonTimestamp = $lastLogonDateProperty
                }
                elseif ($lastLogonDateProperty -is [string]) {
                    try {
                        $lastLogonTimestamp = [datetime]::Parse($lastLogonDateProperty, [System.Globalization.CultureInfo]::InvariantCulture)
                    } catch {
                         try {
                            $lastLogonTimestamp = [datetime]::Parse($lastLogonDateProperty)
                         } catch {
                            $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - Warning: Could not parse LastLogonDate string '$lastLogonDateProperty' for user '$($user.SamAccountName)'. Error: $($_.Exception.Message)`n")
                         }
                    }
                }
                elseif ($lastLogonDateProperty -is [long] -or $lastLogonDateProperty -is [int]) {
                    try {
                        $lastLogonTimestamp = [datetime]::FromFileTime($lastLogonDateProperty)
                    } catch {
                        $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - Warning: Could not convert LastLogonDate FileTime '$lastLogonDateProperty' for user '$($user.SamAccountName)'. Error: $($_.Exception.Message)`n")
                    }
                }

                if ($lastLogonTimestamp -is [datetime]) {
                     if (((Get-Date) - $lastLogonTimestamp).TotalDays -gt 90) { $status = ">90 DAYS" } else { $status = "Recent" }
                } else {
                    $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - Warning: LastLogonDate for user '$($user.SamAccountName)' could not be determined or parsed correctly.`n")
                    $status = "Unknown/Error"
                }
            }
            
            $formattedLastLogonDate = if ($lastLogonTimestamp -is [datetime]) { $lastLogonTimestamp.ToString("yyyy-MM-dd") } else { "" }
            
            $passwordStatus = if ($status -eq "Recent" -and $user.Enabled) { "Already Set" } else { "To Be Reset" }
            
            $rowIndex = $dataGrid.Rows.Add($user.SamAccountName, $passwordStatus, $user.mail, $user.Description, $user.PhysicalDeliveryOfficeName, $user.Enabled, $status, $formattedLastLogonDate)
            $dataGrid.Rows[$rowIndex].Tag = $user.SamAccountName

            if (-not $user.Enabled -or $status -eq "Never" -or $status -eq ">90 DAYS" -or $status -eq "Unknown/Error") {
                $dataGrid.Rows[$rowIndex].DefaultCellStyle.BackColor = $dataGridFlaggedBackColor
                $dataGrid.Rows[$rowIndex].DefaultCellStyle.ForeColor = $dataGridStatusForeColor
                $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - User '$($user.SamAccountName)' flagged for reset/enable.`n")
            } else {
                $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - User '$($user.SamAccountName)' found (Active).`n")
            }
        } catch {
            $errorMessage = $_.Exception.Message
            $rowIndex = $dataGrid.Rows.Add($username, "ERROR", "", "", "", "", "Error", $errorMessage)
            $dataGrid.Rows[$rowIndex].DefaultCellStyle.BackColor = $dataGridErrorBackColor
            $dataGrid.Rows[$rowIndex].DefaultCellStyle.ForeColor = $dataGridStatusForeColor
            $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - Error querying user '$username': $($errorMessage).`n")
        }
        $progressBar.Value++
        $form.Refresh()
    }
    $form.PerformLayout()
    ResizePasswordColumn

    if ($dataGrid.Rows.Count -gt 0) {
        $buttonSave.Enabled = $true; SetButtonStateColors $buttonSave $true
        $usersToAction = $dataGrid.Rows | Where-Object {
            ($_.DefaultCellStyle.BackColor -eq $dataGridFlaggedBackColor) -and 
            ($_.Cells['LogStat'].Value -ne "Process Error") -and 
            ($_.Cells['LogStat'].Value -ne "Error")
        }
        if ($usersToAction -and $usersToAction.Count -gt 0) {
            $buttonEnableReset.Enabled = $true; SetButtonStateColors $buttonEnableReset $true
        } else {
            $buttonEnableReset.Enabled = $false; SetButtonStateColors $buttonEnableReset $false
        }
    } else {
         $buttonSave.Enabled = $false; SetButtonStateColors $buttonSave $false
         $buttonEnableReset.Enabled = $false; SetButtonStateColors $buttonEnableReset $false
    }
    $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - AD query complete. Retrieved $($dataGrid.Rows.Count) results.`n")
})

$buttonEnableReset.Add_Click({
    $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - Starting enable/reset process.`n")
    $form.Cursor = [System.Windows.Forms.Cursors]::WaitCursor
    $buttonEnableReset.Enabled = $false; SetButtonStateColors $buttonEnableReset $false
    $buttonProcess.Enabled = $false; SetButtonStateColors $buttonProcess $false
    $buttonSave.Enabled = $false; SetButtonStateColors $buttonSave $false
    $buttonGeneratePassword.Enabled = $false; SetButtonStateColors $buttonGeneratePassword $false

    $progressBar.Value = 0
    $rowsToProcessQuery = $dataGrid.Rows | Where-Object {
        ($_.DefaultCellStyle.BackColor -eq $dataGridFlaggedBackColor) -and 
        ($_.Cells['LogStat'].Value -ne "Error") -and 
        ($_.Cells['LogStat'].Value -ne "Process Error")
    }
    $progressBar.Maximum = $rowsToProcessQuery.Count
    
    $processedCount = 0
    $usedPasswords = @{}
    $passwordResetDone = @{}

    $targetOUObject = $null
    try {
        $targetOUObject = Get-ADOrganizationalUnit -Identity $targetOU -ErrorAction Stop
    } catch {
        $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - CRITICAL ERROR: Target OU '$targetOU' not found. Aborting process. Error: $($_.Exception.Message)`n")
        $form.Cursor = [System.Windows.Forms.Cursors]::Default
        $buttonProcess.Enabled = $true; SetButtonStateColors $buttonProcess $true
        if ($dataGrid.Rows.Count -gt 0) { $buttonSave.Enabled = $true; SetButtonStateColors $buttonSave $true }
        $usersToAction = $dataGrid.Rows | Where-Object {($_.DefaultCellStyle.BackColor -eq $dataGridFlaggedBackColor) -and ($_.Cells['LogStat'].Value -ne "Process Error") -and ($_.Cells['LogStat'].Value -ne "Error")}
        if ($usersToAction -and $usersToAction.Count -gt 0) { $buttonEnableReset.Enabled = $true; SetButtonStateColors $buttonEnableReset $true }

        $buttonGeneratePassword.Enabled = $true; SetButtonStateColors $buttonGeneratePassword $true
        return
    }
    $targetOUDN = $targetOUObject.DistinguishedName
    $targetOUFriendlyName = $targetOUObject.Name

    foreach ($row in $dataGrid.Rows) {
        if ($row.DefaultCellStyle.BackColor -ne $dataGridFlaggedBackColor `
            -or $row.Cells['LogStat'].Value -eq "Error" `
            -or $row.Cells['LogStat'].Value -eq "Process Error") {
            if ($row.Tag -and $passwordResetDone.ContainsKey($row.Tag) -and $row.Cells['Password'].Value -ne "ERROR") {
                 $row.Cells['Password'].Value = "SET PREVIOUSLY"
                 $row.DefaultCellStyle.BackColor = $dataGridSetPreviouslyBackColor
                 $row.DefaultCellStyle.ForeColor = $dataGridStatusForeColor
            }
            continue
        }

        $usernameToProcess = $row.Tag
        if (-not $usernameToProcess) {
            $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - Skipping row with no username tag.`n")
            continue
        }

        if ($passwordResetDone.ContainsKey($usernameToProcess)) {
            $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - User '$usernameToProcess' already processed in this batch. Updating row.`n")
            $row.Cells['Password'].Value = "SET PREVIOUSLY"
            $row.DefaultCellStyle.BackColor = $dataGridSetPreviouslyBackColor
            $row.DefaultCellStyle.ForeColor = $dataGridStatusForeColor
            $row.Cells['LogStat'].Value = "Processed (Duplicate)"
            continue
        }
        
        $progressBar.Value++
        $form.Refresh()

        try {
            $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - Processing '$usernameToProcess': Enable, Move, and Reset Password.`n")
            $currentUser = Get-ADUser -Identity $usernameToProcess -Properties Enabled, DistinguishedName, SamAccountName -ErrorAction Stop

            if (-not $currentUser.Enabled) {
                 $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - Enabling user '$usernameToProcess'.`n")
                 Set-ADUser -Identity $currentUser -Enabled $true -ErrorAction Stop
                 $row.Cells['Enabled'].Value = $true
                 $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - Re-fetching user '$usernameToProcess' after enabling.`n")
                 $currentUser = Get-ADUser -Identity $currentUser.SamAccountName -Properties Enabled, DistinguishedName, SamAccountName -ErrorAction Stop
            } else {
                 $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - User '$usernameToProcess' is already enabled.`n")
            }

            $currentUserParentPath = ($currentUser.DistinguishedName -split ',', 2)[1]

            if ($currentUserParentPath -ne $targetOUDN) {
                 $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - Moving user '$usernameToProcess' from '$currentUserParentPath' to '$targetOUDN'.`n")
                 Move-ADObject -Identity $currentUser -TargetPath $targetOUDN -ErrorAction Stop
                 $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - Re-fetching user '$usernameToProcess' after moving.`n")
                 $currentUser = Get-ADUser -Identity $currentUser.SamAccountName -Properties Enabled, DistinguishedName, SamAccountName -ErrorAction Stop
            } else {
                 $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - User '$usernameToProcess' is already in OU: '$targetOUFriendlyName'. No move action performed.`n")
            }
            
            $newPassword = Generate-UniquePassword $usedPasswords.Keys $adjectiveBank $nounBank $verbBank
            while ($usedPasswords.ContainsKey($newPassword)) {
                $newPassword = Generate-UniquePassword $usedPasswords.Keys $adjectiveBank $nounBank $verbBank
            }
            $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - Setting new password for '$usernameToProcess' and forcing change at next logon.`n")
            Set-ADAccountPassword -Identity $currentUser -NewPassword (ConvertTo-SecureString $newPassword -AsPlainText -Force) -Reset -ErrorAction Stop
            Set-ADUser -Identity $currentUser -ChangePasswordAtLogon $true -ErrorAction Stop
            
            $row.Cells['Password'].Value = $newPassword
            $usedPasswords[$newPassword] = $true
            $passwordResetDone[$usernameToProcess] = $true
            
            $row.DefaultCellStyle.BackColor = $dataGridProcessedOKBackColor
            $row.DefaultCellStyle.ForeColor = $dataGridStatusForeColor
            $row.Cells['LogStat'].Value = "Processed OK"
            $processedCount++
            $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - Successfully processed '$usernameToProcess'.`n")

        } catch {
             $errorMessage = $_.Exception.Message
             $row.Cells['Password'].Value = "ERROR"
             $row.DefaultCellStyle.BackColor = $dataGridErrorBackColor
             $row.DefaultCellStyle.ForeColor = $dataGridStatusForeColor
             $row.Cells['LogStat'].Value = "Process Error"
             $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - Error processing '$usernameToProcess': $($errorMessage).`n")
        }
    }

    ResizePasswordColumn
    $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - Enable/reset process complete. $processedCount users successfully processed.`n")
    $form.Cursor = [System.Windows.Forms.Cursors]::Default

    $buttonProcess.Enabled = $true; SetButtonStateColors $buttonProcess $true
    if ($dataGrid.Rows.Count -gt 0) { $buttonSave.Enabled = $true; SetButtonStateColors $buttonSave $true }
    
    $usersStillToAction = $dataGrid.Rows | Where-Object {
        ($_.DefaultCellStyle.BackColor -eq $dataGridFlaggedBackColor) -and 
        ($_.Cells['LogStat'].Value -ne "Process Error") -and 
        ($_.Cells['LogStat'].Value -ne "Error") -and
        (-not $passwordResetDone.ContainsKey($_.Tag))
    }
    if ($usersStillToAction -and $usersStillToAction.Count -gt 0) {
        $buttonEnableReset.Enabled = $true; SetButtonStateColors $buttonEnableReset $true
    } else {
        $buttonEnableReset.Enabled = $false; SetButtonStateColors $buttonEnableReset $false
    }
    $buttonGeneratePassword.Enabled = $true; SetButtonStateColors $buttonGeneratePassword $true
})

$buttonSave.Add_Click({
    $saveFileDialog = New-Object System.Windows.Forms.SaveFileDialog
    $saveFileDialog.Filter = "CSV files (*.csv)|*.csv|All files (*.*)|*.*"
    $saveFileDialog.Title = "Save Results"
    $saveFileDialog.FileName = "ADResults-$(Get-Date -UFormat %Y-%m-%d).csv"
    $saveFileDialog.InitialDirectory = [System.Environment]::GetFolderPath('MyDocuments')

    if ($saveFileDialog.ShowDialog() -eq [System.Windows.Forms.DialogResult]::OK) {
        $csvPath = $saveFileDialog.FileName
        $outputList = @()
        try {
            $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - Saving results to '$csvPath'.`n")
            $form.Cursor = [System.Windows.Forms.Cursors]::WaitCursor
            $buttonSave.Enabled = $false; SetButtonStateColors $buttonSave $false
            $buttonProcess.Enabled = $false; SetButtonStateColors $buttonProcess $false
            $buttonEnableReset.Enabled = $false; SetButtonStateColors $buttonEnableReset $false
            $buttonGeneratePassword.Enabled = $false; SetButtonStateColors $buttonGeneratePassword $false

            foreach ($row in $dataGrid.Rows) {
                 if ([string]::IsNullOrWhiteSpace($row.Cells['Username'].Value) -and -not $row.Tag) {
                     continue
                 }
                $rowData = [ordered]@{}
                foreach ($col in $dataGrid.Columns) {
                    $cellValue = if ($row.Cells[$col.Name].Value -ne $null) { $row.Cells[$col.Name].Value } else { "" }
                    $rowData[$col.Name] = [string]$cellValue
                }
                $outputList += (New-Object PSObject -Property $rowData)
            }

            if ($outputList.Count -gt 0) {
                 $outputList | Export-Csv -Path $csvPath -NoTypeInformation -UseCulture:$false -Encoding UTF8
                 $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - Data successfully saved to '$csvPath'. Exported $($outputList.Count) data rows.`n")
            } else {
                 $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - No data rows to export.`n")
            }
        } catch {
             $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - Error saving data: $($_.Exception.Message).`n")
        } finally {
            $form.Cursor = [System.Windows.Forms.Cursors]::Default
            $buttonProcess.Enabled = $true; SetButtonStateColors $buttonProcess $true
            if ($dataGrid.Rows.Count -gt 0) { $buttonSave.Enabled = $true; SetButtonStateColors $buttonSave $true }
            $usersToAction = $dataGrid.Rows | Where-Object {($_.DefaultCellStyle.BackColor -eq $dataGridFlaggedBackColor) -and ($_.Cells['LogStat'].Value -ne "Process Error") -and ($_.Cells['LogStat'].Value -ne "Error")}
            if ($usersToAction -and $usersToAction.Count -gt 0) { $buttonEnableReset.Enabled = $true; SetButtonStateColors $buttonEnableReset $true } else {$buttonEnableReset.Enabled = $false; SetButtonStateColors $buttonEnableReset $false}

            $buttonGeneratePassword.Enabled = $true; SetButtonStateColors $buttonGeneratePassword $true
        }
    } else {
        $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - Save operation cancelled by user.`n")
    }
})

$buttonGeneratePassword.Add_Click({
    try {
        if (-not $adjectiveBank -or -not $nounBank -or -not $verbBank `
            -or $adjectiveBank.Count -eq 0 -or $nounBank.Count -eq 0 -or $verbBank.Count -eq 0) {
             $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - Error: Password generation banks (adjectiveBank, nounBank, verbBank) are not defined or empty.`n")
            Return
        }

        $generatedPwd = Generate-UniquePassword @() $adjectiveBank $nounBank $verbBank
        $textBoxGeneratedPassword.Text = $generatedPwd
        $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - Generated new random password.`n")

        $usernameForCopy = $textBoxGenPasswordUser.Text
        $passwordToCopy = ""
        if (-not [string]::IsNullOrWhiteSpace($usernameForCopy)) {
            $passwordToCopy = "Username: $($usernameForCopy)`n"
        }
        $passwordToCopy += "Password: $($generatedPwd)"
        
        [System.Windows.Forms.Clipboard]::SetText($passwordToCopy)
        $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - Details copied to clipboard.`n")

        $passwordTextboxRightBoundary = $form.ClientSize.Width - 20 
        ResizeTextBoxToContent -TextBox $textBoxGeneratedPassword `
                               -RightBoundaryX $passwordTextboxRightBoundary `
                               -MinimumWidth $textBoxGenPasswordUser.Width `
                               -MaximumWidth ($form.ClientSize.Width - $textBoxGeneratedPassword.Left - 20)
		
    } catch {
        $textBoxStatusLog.AppendText("$(Get-Date -Format 'HH:mm:ss') - Error generating password: $($_.Exception.Message).`n")
    }
})
